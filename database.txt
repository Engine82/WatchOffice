TABLES:

CREATE TABLE users (
id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
username TEXT NOT NULL,
hash TEXT NOT NULL,
rank TEXT,
platoon INTEGER,
active INTEGER NOT NULL,
elligible INTEGER NOT NULL,
tag_flipped INT,
created_at DATETIME DEFAULT (CURRENT_TIMESTAMP) NOT NULL
);

CREATE TABLE hiring (
id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
hiring_id INTEGER NOT NULL,
platoon INTEGER NOT NULL,
rank TEXT NOT NULL,
day INTEGER NOT NULL,
time TEXT NOT NULL,
member_out INTEGER NOT NULL,
member_covering INTEGER NOT NULL,
created_at DATETIME DEFAULT(CURRENT_TIMESTAMP) NOT NULL,
FOREIGN KEY (member_out) REFERENCES users(id),
FOREIGN KEY (member_covering) REFERENCES users(id)
);
